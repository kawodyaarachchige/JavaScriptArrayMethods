01 concat()

Joins two or more arrays and returns a new array.

const array1 = [1, 2, 3];
const array2 = [4, 5, 6];
const result = array1.concat(array2);

console.log(result); // Output: [1, 2, 3, 4, 5, 6]

02 copyWithin()

Copies part of an array to another location in the same array and returns it.

const array = [1, 2, 3, 4, 5];
array.copyWithin(0, 3);

console.log(array); // Output: [4, 5, 3, 4, 5]

array.copyWithin(target, start[, end])
target: The index at which to copy the sequence to.
start: The index at which to start copying elements from.
end (optional): The index at which to end copying elements from (not included).

03 entries ()

Returns a new array iterator object that contains the key/value pairs for each index in the array.

const array = ['a', 'b', 'c'];  //Step 1: Create an array
const iterator = array.entries();   //Step 2: Get the iterator from array entries

// Step 3: Use for...of loop to iterate over the iterator

for (let [index, element] of iterator) {
  console.log(index, element);
}
// Output:
// 0 'a'
// 1 'b'
// 2 'c'

04 every()

Tests whether all elements in the array pass the test implemented by the provided function.

const array = [1, 2, 3, 4, 5];
const allPositive = array.every(num => num > 0);

console.log(allPositive); // Output: true

05 fill()

Fills all the elements of an array from a start index to an end index with a static value.

const array = [1, 2, 3, 4, 5];
array.fill(0, 2, 4); // Fill with 0 from index 2 to index 4 (exclusive)

console.log(array); // Output: [1, 2, 0, 0, 5]

06 filter()

Creates a new array with all elements that pass the test implemented by the provided function.

const array = [1, 2, 3, 4, 5];
const filtered = array.filter(num => num > 2);

console.log(filtered); // Output: [3, 4, 5]

07 find()

Returns the value of the first element in the array that satisfies the provided testing function.

const array = [1, 2, 3, 4, 5];
const found = array.find(num => num > 3);

console.log(found); // Output: 4

08 findIndex()

Returns the index of the first element in the array that satisfies the provided testing function.

const array = [1, 2, 3, 4, 5];
const foundIndex = array.findIndex(num => num > 3);

console.log(foundIndex); // Output: 3

09 flat()

Creates a new array with all sub-array elements concatenated into it recursively up to the specified depth.

const array = [1, 2, [3, 4, [5, 6]]];
const flatArray = array.flat(2);

console.log(flatArray); // Output: [1, 2, 3, 4, 5, 6]

10 flatMap()

First maps each element using a mapping function, then flattens the result into a new array.

const array = [1, 2, 3, 4];
const flatMapped = array.flatMap(num => [num * 2]);

console.log(flatMapped); // Output: [2, 4, 6, 8]

11 forEach()

Executes a provided function once for each array element.

const array = [1, 2, 3];
array.forEach(num => console.log(num));

// Output:
// 1
// 2
// 3

12 Array.from()

Converts an array-like object to an array.

const array = Array.from('hello');
console.log(array); // Output: ['h', 'e', 'l', 'l', 'o']

13 includes()

Determines whether an array includes a certain value among its entries.

const array = [1, 2, 3];
const hasTwo = array.includes(2);

console.log(hasTwo); // Output: true

14 indexOf()

Returns the first index at which a given element can be found in the array, or -1 if it is not present.

const array = [1, 2, 3, 4, 5];
const index = array.indexOf(3);

console.log(index); // Output: 2

15 isArray()

Determines whether the passed value is an Array.

const array = [1, 2, 3];
const isArray = Array.isArray(array);

console.log(isArray); // Output: true

16 join()

Joins all elements of an array into a string.

const array = ['a', 'b', 'c'];
const joined = array.join('-');

console.log(joined); // Output: 'a-b-c'

17keys()

Returns a new Array Iterator object that contains the keys for each index in the array.

const array = ['a', 'b', 'c'];
const keys = array.keys();

for (let key of keys) {
  console.log(key); // Output: 0, 1, 2
}

18 lastIndexOf()

Returns the last index at which a given element can be found in the array, or -1 if it is not present.

const array = [1, 2, 3, 4, 5];
const index = array.lastIndexOf(3);

console.log(index); // Output: 2

19 map()

Creates a new array with the results of calling a provided function on every element in the calling array.

const array = [1, 2, 3];
const mapped = array.map(num => num * 2);

console.log(mapped); // Output: [2, 4, 6]

20 Array.of()

Creates a new Array instance with a variable number of arguments, regardless of number or type of the arguments.

const array = Array.of(1, 2, 3);

console.log(array); // Output: [1, 2, 3]

21 pop()

Removes the last element from an array and returns that element. This method changes the length of the array.

const array = [1, 2, 3];
const popped = array.pop();

console.log(popped); // Output: 3
console.log(array); // Output: [1, 2]

22 push()

Adds one or more elements to the end of an array and returns the new length of the array.

const array = [1, 2, 3];
const length = array.push(4, 5);

console.log(length); // Output: 5
console.log(array); // Output: [1, 2, 3, 4, 5]

23 reduce()

Applies a function against an accumulator and each element in the array (from left to right) to reduce it to a single value

const array = [1, 2, 3, 4];
const sum = array.reduce((acc, num) => acc + num, 0); // Reduce the array to a single value (sum)

console.log(sum); // Output: 10

24 reduceRight()

Applies a function against an accumulator and each element in the array (from right to left) to reduce it to a single value.

const array = [1, 2, 3, 4];
const sum = array.reduceRight((acc, num) => acc + num, 0);

console.log(sum); // Output: 10

25 reverse()

Reverses an array in place. The first array element becomes the last, and the last array element becomes the first.

const array = [1, 2, 3, 4];
array.reverse();
    console.log(array); // Output: [4, 3, 2, 1]

26 shift()

Removes the first element from an array and returns that removed element. This method changes the length of the array.

const array = [1, 2, 3];
const shifted = array.shift();

console.log(shifted); // Output: 1
console.log(array); // Output: [2, 3]

27 slice()

Extracts a section of an array and returns a new array.

const array = [1, 2, 3, 4, 5];
const sliced = array.slice(2, 4);

console.log(sliced); // Output: [3, 4]

28 some()

Determines whether at least one element in the array passes the test implemented by the provided function.

const array = [1, 2, 3, 4, 5];
const result = array.some(num => num % 2 === 0);

console.log(result); // Output: true

29 sort()

Sorts the elements of an array in place and returns the sorted array.

const array = [3, 2, 1];
array.sort();
console.log(array); // Output: [1, 2, 3]

30 splice()

Removes elements from an array and, if necessary, inserts new elements in their place, returning the deleted elements.

const array = [1, 2, 3, 4, 5];
const spliced = array.splice(2, 2);

console.log(spliced); // Output: [3, 4]
console.log(array); // Output: [1, 2, 5]

31 toLocaleString()

Returns a string representing the elements of the array. The elements are converted to strings using their toLocaleString methods.

const array = [1, 'a', new Date()];
const localeString = array.toLocaleString();

console.log(localeString); // Output: "1,a,7/7/2024, 10:11:00 AM"

32 toString ()

Returns a string representing the specified array and its element.

const array = [1, 2, 'a', 'b'];
const string = array.toString();

console.log(string); // Output: "1,2,a,b"

33 unshift()

Adds one or more elements to the beginning of an array and returns the new length of the array.

const array = [2, 3];
const newLength = array.unshift(1);

console.log(newLength); // Output: 3
console.log(array); // Output: [1, 2, 3]

34 values()

Returns a new Array Iterator object that contains the values for each index in the array.

const array = ['a', 'b', 'c'];
const iterator = array.values();

for (let value of iterator) {
  console.log(value);
}
// Output:
// 'a'
// 'b'
// 'c'
